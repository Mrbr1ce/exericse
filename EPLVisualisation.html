# create a ColumnDataSource from our df: source
source=ColumnDataSource(data)

# create the figure: p
p=figure(tools=['hover'])

# in figure p, display the points as circles
p.circle('market_value', 'page_views', source=source, color='blue', size=5, alpha=0.5)

# specify the hover tool
hover=HoverTool(tooltips=[('Name','@name'),('Team','@club'),('Age', '@age')])

p.add_tools(hover)

show(p)
